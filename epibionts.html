Creating Full Clean Dataset

full_cover_clean <- cover_clean_names |>
  filter(site != "Asilomar 3" &
         site != "Cannery Row 3" &
         site != "Great Tidepool 2" &
         site != "Pacific Grove 1") |>                                        # removing site with no
  unite(code, 1:3, sep = "-", remove = TRUE, na.rm = FALSE) |>                # creating code variable
  group_by(code) |>                                                           # grouping to help with later code
  mutate(algae_cover = 
           sum(c_across(filamentous_turf:surfgrass), na.rm = T)) |>           # adding cover of all algae
  mutate(sessile_invert_cover = 
           sum(c_across(pink_volcano_barnacle:serpulidae), na.rm = T)) |>     # adding cover of all sessile inverts
  dplyr::select(code, height, bare_rock, algae_cover, sessile_invert_cover)   # selecting only needed columns
  
epibiosis_clean <- full_cover_clean |>
  left_join(counts_clean, by = "code") |>                                     # combining with counts by code
  mutate(across(is.numeric, ~replace_na(., 0)))                               # replacing NAs with 0s
